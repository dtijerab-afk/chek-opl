📌 SQL para crear la tabla en Supabase:

CREATE TABLE evaluaciones (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  zona text NOT NULL, tienda text NOT NULL,
  operador text NOT NULL, administrador text,
  fecha date NOT NULL, datos jsonb,
  score_pdv numeric, score_bodegas numeric,
  score_opl numeric, score_admin numeric,
  resultado numeric, clasificacion text,
  firma_coord text, firma_admin text,
  created_at timestamptz DEFAULT now()
);
ALTER TABLE evaluaciones ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all" ON evaluaciones
  FOR ALL USING (true) WITH CHECK (true);
